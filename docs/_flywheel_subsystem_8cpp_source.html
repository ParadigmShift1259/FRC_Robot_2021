99 void FlywheelSubsystem::CalculateRPM()
100 {
101  // Ignore PIDF feedforward and substitute WPILib's SimpleMotorFeedforward class
102  double FF = m_flywheelFF.Calculate(m_setpoint / kSecondsPerMinute * 1_mps).to<double>();
103  m_flywheelPID.SetFF(0);
104  // Choose "Maintain" PID set, or slot 1, if we're near the setpoint
105  int pidslot = IsAtMaintainPID() ? 1 : 0;
106  m_flywheelPID.SetReference(m_setpoint, ControlType::kVelocity, pidslot, FF);
107 }
