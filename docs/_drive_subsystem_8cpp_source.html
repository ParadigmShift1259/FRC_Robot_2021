116 void DriveSubsystem::RotationDrive(meters_per_second_t xSpeed
117  , meters_per_second_t ySpeed
118  , radian_t rot
119  , bool fieldRelative)
120 {
121  double error = rot.to<double>() - m_gyro->GetHeadingAsRot2d().Radians().to<double>();
122  double desiredSet = Util::NegPiToPiRads(error);
123  double max = DriveConstants::kMaxAbsoluteRotationSpeed;
124  double maxTurn = DriveConstants::kMaxAbsoluteTurnableSpeed;
125 
126  #ifdef TUNE_ROTATION
127  double P = SmartDashboard::GetNumber("T_D_RP", 0);
128  double I = SmartDashboard::GetNumber("T_D_RI", 0);
129  double D = SmartDashboard::GetNumber("T_D_RD", 0);
130  double m = SmartDashboard::GetNumber("T_D_RMax", 0);
131  double mTurn = SmartDashboard::GetNumber("T_D_RTMax", 0);
132 
133  m_rotationPIDController.SetP(P);
134  m_rotationPIDController.SetI(I);
135  m_rotationPIDController.SetD(D);
136  max = m;
137  maxTurn = mTurn;
138  #endif
139 
140  double desiredTurnRate = m_rotationPIDController.Calculate(0, desiredSet);
141 
142  double currentTurnRate = m_gyro->GetTurnRate() * wpi::math::pi / 180;
143 
144  // Prevent sharp turning if already fast going in a direction
145  if ((abs(currentTurnRate) >= maxTurn) && (signbit(desiredTurnRate) != signbit(currentTurnRate)))
146  desiredTurnRate *= -1.0;
147 
148  // Power limiting
149  if (abs(desiredTurnRate) > max)
150  desiredTurnRate = signbit(desiredTurnRate) ? max * -1.0 : max;
151 
152  // if (!m_rotationPIDController.AtSetpoint())
153  Drive(xSpeed, ySpeed, radians_per_second_t(desiredTurnRate), fieldRelative);
154  // else
155  // Drive(xSpeed, ySpeed, radians_per_second_t(0), fieldRelative);
156 }

174 void DriveSubsystem::HeadingDrive(meters_per_second_t xSpeed
175  , meters_per_second_t ySpeed
176  , radians_per_second_t rot
177  , bool fieldRelative)
178 {
179  // SmartDashboard::PutBoolean("D_D_RotationInput", m_rotationalInput);
180  // SmartDashboard::PutNumber("D_D_lastHeading", m_lastHeading);
181  if (xSpeed.to<double>() == 0 && ySpeed.to<double>() == 0 && rot.to<double>() == 0)
182  {
183  Drive(xSpeed, ySpeed, rot, fieldRelative);
184  return;
185  }
186 
187  if (rot.to<double>() == 0 && m_rotationalInput)
188  {
189  m_rotationalInput = false;
190  UpdateLastHeading();
191  }
192  else
193  if (rot.to<double>() != 0)
194  m_rotationalInput = true;
195 
196  if (!m_rotationalInput && (xSpeed.to<double>() != 0 || ySpeed.to<double>() != 0))
197  RotationDrive(xSpeed, ySpeed, radian_t(m_lastHeading), fieldRelative);
198  else
199  Drive(xSpeed, ySpeed, rot, fieldRelative);
200 }

202 void DriveSubsystem::Drive(meters_per_second_t xSpeed
203  , meters_per_second_t ySpeed
204  , radians_per_second_t rot
205  , bool fieldRelative)
206 {
207  rot = radians_per_second_t(rot.to<double>() * AutoConstants::kTeleMaxAngularSpeed.to<double>());
208 
209  frc::ChassisSpeeds chassisSpeeds;
210  if (fieldRelative)
211  chassisSpeeds = frc::ChassisSpeeds::FromFieldRelativeSpeeds(xSpeed, ySpeed, rot, m_gyro->GetHeadingAsRot2d());
212  else
213  chassisSpeeds = frc::ChassisSpeeds{xSpeed, ySpeed, rot};
214 
215  auto states = kDriveKinematics.ToSwerveModuleStates(chassisSpeeds);
216 
217  kDriveKinematics.NormalizeWheelSpeeds(&states, AutoConstants::kMaxSpeed);
218 
219  #ifdef TUNE_MODULES
220  states[eFrontLeft].angle = frc::Rotation2d(radian_t(SmartDashboard::GetNumber("T_D_MFL", 0.0)));
221  states[eFrontRight].angle = frc::Rotation2d(radian_t(SmartDashboard::GetNumber("T_D_MFR", 0.0)));
222  states[eRearRight].angle = frc::Rotation2d(radian_t(SmartDashboard::GetNumber("T_D_MRR", 0.0)));
223  states[eRearLeft].angle = frc::Rotation2d(radian_t(SmartDashboard::GetNumber("T_D_MRL", 0.0)));
224  states[eFrontLeft].speed = SmartDashboard::GetNumber("T_D_MFLV", 0.0) * 1_mps;
225  states[eFrontRight].speed = SmartDashboard::GetNumber("T_D_MFRV", 0.0) * 1_mps;
226  states[eRearRight].speed = SmartDashboard::GetNumber("T_D_MRRV", 0.0) * 1_mps;
227  states[eRearLeft].speed = SmartDashboard::GetNumber("T_D_MRLV", 0.0) * 1_mps;
228  #endif
229 
230  m_frontLeft.SetDesiredState(states[eFrontLeft]);
231  m_frontRight.SetDesiredState(states[eFrontRight]);
232  m_rearLeft.SetDesiredState(states[eRearLeft]);
233  m_rearRight.SetDesiredState(states[eRearRight]);
234 }
