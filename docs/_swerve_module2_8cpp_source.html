92  // Find absolute encoder and NEO encoder positions
93  //EncoderToRadians();
94  double currentPosition = m_turnRelativeEncoder.GetPosition();
95 
96  // Calculate new turn position given current Neo position, current absolute encoder position, and desired state position
97  bool bOutputReverse = false;
98  double minTurnRads = MinTurnRads(currentPosition, state.angle.Radians().to<double>(), bOutputReverse);
99  double direction = 1.0; // Sent to TalonFX
100  if (bOutputReverse)
101  direction = -1.0;
102 
103  // Set position reference of turnPIDController
104  double newPosition = currentPosition + minTurnRads;
105 
106  if (state.speed != 0_mps) {
107  #ifdef DISABLE_DRIVE
108  m_driveMotor.Set(TalonFXControlMode::Velocity, 0.0);
109  #else
110  m_driveMotor.Set(TalonFXControlMode::Velocity, direction * CalcTicksPer100Ms(state.speed));
111  #endif
112  }
113  else {
114  m_driveMotor.Set(TalonFXControlMode::PercentOutput, 0.0);
115  }
116 
117 
118  // Set the angle unless module is coming to a full stop
119  if (state.speed.to<double>() != 0.0)
120  m_turnPIDController.SetReference(newPosition, rev::ControlType::kPosition);
121 
122 
123  SmartDashboard::PutNumber("D_SM_SetpointMPS " + m_name, state.speed.to<double>());
124  // SmartDashboard::PutNumber("D_SM_Error " + m_name, newPosition - m_turnRelativeEncoder.GetPosition());

132 void SwerveModule2::EncoderToRadians()
133 {
134  double pulseWidth = m_pulseWidthCallback(m_pwmChannel);
135  // Pulse Width per rotation is not equal for all encoders. Some are 0 - 3865, some are 0 - 4096
136  m_absAngle = fmod((pulseWidth - m_offset) * DriveConstants::kPulseWidthToRadians + Math::kTau, Math::kTau);
137  SmartDashboard::PutNumber("D_SM_PW " + m_name, pulseWidth);
138  SmartDashboard::PutNumber("D_SM_AA " + m_name, m_absAngle);
139  // Convert CW to CCW? m_absAngle = Math::kTau - m_absAngle;
140 }
